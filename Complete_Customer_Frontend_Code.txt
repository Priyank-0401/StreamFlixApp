import axios from 'axios';
import {
  CustomerProfileDTO,
  PlanDTO,
  AddOnDTO,
  NotificationDTO,
  SubscriptionDTO,
  UsageRecordDTO,
  InvoiceDTO,
  PaymentDTO,
  CreditNoteDTO,
  CouponDTO,
  PaymentMethodDTO,
  SupportMessageDTO,
  FAQ,
  CreateSubscriptionRequest,
  UpgradeSubscriptionRequest,
  PauseSubscriptionRequest,
  AddPaymentMethodRequest,
  DashboardStats
} from '../types/customer';

const API_BASE_URL = 'http://localhost:8765/api/customer';

// Create axios instance with auth header
const apiClient = axios.create({
  baseURL: API_BASE_URL,
  headers: {
    'Content-Type': 'application/json',
  },
});

// Add request interceptor to attach token
apiClient.interceptors.request.use((config) => {
  const token = localStorage.getItem('token');
  if (token) {
    config.headers.Authorization = `Bearer ${token}`;
  }
  return config;
});

// Add response interceptor for error handling
apiClient.interceptors.response.use(
  (response) => response,
  (error) => {
    if (error.response?.status === 401) {
      localStorage.removeItem('token');
      window.location.href = '/login';
    }
    return Promise.reject(error);
  }
);

export const CustomerService = {
  // Profile
  getProfile: () => apiClient.get<CustomerProfileDTO>('/profile').then(r => r.data),
  updateProfile: (data: Partial<CustomerProfileDTO>) => 
    apiClient.put<CustomerProfileDTO>('/profile', data).then(r => r.data),

  // Plans & Add-ons
  getAvailablePlans: () => apiClient.get<PlanDTO[]>('/plans').then(r => r.data),
  getAvailableAddOns: () => apiClient.get<AddOnDTO[]>('/addons').then(r => r.data),

  // Notifications
  getNotifications: () => apiClient.get<NotificationDTO[]>('/notifications').then(r => r.data),
  markNotificationAsRead: (notificationId: number) => 
    apiClient.post(`/notifications/${notificationId}/read`).then(r => r.data),

  // Dashboard
  getDashboardStats: () => apiClient.get<DashboardStats>('/dashboard').then(r => r.data),

  // Subscription
  getCurrentSubscription: () => 
    apiClient.get<SubscriptionDTO | null>('/subscription').then(r => r.data),
  createSubscription: (data: CreateSubscriptionRequest) => 
    apiClient.post<SubscriptionDTO>('/subscription', data).then(r => r.data),
  upgradeSubscription: (data: UpgradeSubscriptionRequest) => 
    apiClient.put<SubscriptionDTO>('/subscription/upgrade', data).then(r => r.data),
  cancelSubscription: (atPeriodEnd: boolean) => 
    apiClient.post('/subscription/cancel', null, { params: { atPeriodEnd } }).then(r => r.data),
  pauseSubscription: (data: PauseSubscriptionRequest) => 
    apiClient.post<SubscriptionDTO>('/subscription/pause', data).then(r => r.data),
  resumeSubscription: () => apiClient.post<SubscriptionDTO>('/subscription/resume').then(r => r.data),
  addAddOn: (addonId: number) => 
    apiClient.post(`/subscription/addons/${addonId}`).then(r => r.data),
  removeAddOn: (addonId: number) => 
    apiClient.delete(`/subscription/addons/${addonId}`).then(r => r.data),
  getMeteredUsage: (startDate?: string, endDate?: string) => 
    apiClient.get<UsageRecordDTO[]>('/subscription/usage', { params: { startDate, endDate } }).then(r => r.data),

  // Billing
  getInvoices: (status?: string, from?: string, to?: string) => 
    apiClient.get<InvoiceDTO[]>('/invoices', { params: { status, from, to } }).then(r => r.data),
  getInvoiceDetail: (invoiceId: number) => 
    apiClient.get<InvoiceDTO>(`/invoices/${invoiceId}`).then(r => r.data),
  downloadInvoicePdf: (invoiceId: number) => 
    apiClient.get(`/invoices/${invoiceId}/pdf`, { responseType: 'blob' }).then(r => r.data),
  getPayments: () => apiClient.get<PaymentDTO[]>('/payments').then(r => r.data),
  getCreditNotes: () => apiClient.get<CreditNoteDTO[]>('/credit-notes').then(r => r.data),
  applyCoupon: (code: string) => apiClient.post<CouponDTO>('/coupons/apply', { code }).then(r => r.data),

  // Payment Methods
  getPaymentMethods: () => apiClient.get<PaymentMethodDTO[]>('/payment-methods').then(r => r.data),
  addPaymentMethod: (data: AddPaymentMethodRequest) => 
    apiClient.post<PaymentMethodDTO>('/payment-methods', data).then(r => r.data),
  setDefaultPaymentMethod: (paymentMethodId: number) => 
    apiClient.put(`/payment-methods/${paymentMethodId}/default`).then(r => r.data),
  deletePaymentMethod: (paymentMethodId: number) => 
    apiClient.delete(`/payment-methods/${paymentMethodId}`).then(r => r.data),

  // Support
  getFAQs: () => apiClient.get<FAQ[]>('/support/faqs').then(r => r.data),
  sendSupportMessage: (data: SupportMessageDTO) => 
    apiClient.post('/support/messages', data).then(r => r.data),
};

    const request: AddPaymentMethodRequest = {
      paymentType,
      isDefault,
    };

    if (paymentType === 'CARD') {
      if (!cardNumber || cardNumber.length < 16) {
        alert('Please enter a valid card number');
        return;
      }
      request.cardNumber = cardNumber;
      request.expiryMonth = parseInt(expiryMonth);
      request.expiryYear = parseInt(expiryYear);
    } else {
      if (!upiId) {
        alert('Please enter UPI ID');
        return;
      }
      request.upiId = upiId;
    }

    try {
      await CustomerService.addPaymentMethod(request);
      loadPaymentMethods();
      setShowAddModal(false);
      resetForm();
    } catch (error) {
      alert('Failed to add payment method');
    }
  };

  const handleSetDefault = async (id: number) => {
    try {
      await CustomerService.setDefaultPaymentMethod(id);
      loadPaymentMethods();
    } catch (error) {
      alert('Failed to set default payment method');
    }
  };

  const handleDelete = async (id: number) => {
    if (!confirm('Are you sure you want to remove this payment method?')) return;
    try {
      await CustomerService.deletePaymentMethod(id);
      loadPaymentMethods();
    } catch (error: any) {
      alert(error.response?.data?.message || 'Failed to delete payment method');
    }
  };

  const resetForm = () => {
    setPaymentType('CARD');
    setCardNumber('');
    setExpiryMonth('');
    setExpiryYear('');
    setUpiId('');

  const getCardIcon = (brand: string | null) => {

  };

  if (loading) {
    return (
      <div className="flex items-center justify-center h-64">
        <div className="animate-spin rounded-full h-8 w-8 border-b-2 border-blue-600"></div>
      </div>
    );

  return (

      <div className="flex items-center justify-between">

                {method.paymentType === 'CARD' ? (
                  getCardIcon(method.cardBrand)
                ) : (
                  <Smartphone className="w-8 h-8 text-green-600" />
                )}
                <div>
                  <p className="font-semibold">
                    {method.paymentType === 'CARD'
                      ? `${method.cardBrand || 'Card'} •••• ${method.cardLast4}`
                      : `UPI: ${method.upiId}`}
                  </p>
                  {method.paymentType === 'CARD' && (
                    <p className="text-sm text-gray-500">
                      Expires {method.expiryMonth}/{method.expiryYear}
                    </p>

                  {method.isDefault && (

import React from 'react';
import { BrowserRouter, Routes, Route, Navigate } from 'react-router-dom';
import { CustomerLayout } from './components/customer/layout/CustomerLayout';
import { OverviewPage } from './components/customer/pages/OverviewPage';
import { SubscriptionPage } from './components/customer/pages/SubscriptionPage';
import { BillingPage } from './components/customer/pages/BillingPage';
import { PaymentMethodsPage } from './components/customer/pages/PaymentMethodsPage';
import { ProfilePage } from './components/customer/pages/ProfilePage';
import { SupportPage } from './components/customer/pages/SupportPage';

// Protected Route wrapper component
const ProtectedRoute: React.FC<{ children: React.ReactNode }> = ({ children }) => {
  const token = localStorage.getItem('token');
  if (!token) {
    return <Navigate to="/login" replace />;
  }
  return <>{children}</>;
};

function App() {
  return (
    <BrowserRouter>
      <Routes>
        {/* Public Routes */}
        <Route path="/login" element={<div>Login Page (To be implemented)</div>} />
        
        {/* Customer Dashboard Routes */}
        <Route
          path="/dashboard"
          element={
            <ProtectedRoute>
              <CustomerLayout />
            </ProtectedRoute>
          }
        >
          <Route index element={<OverviewPage />} />
          <Route path="subscription" element={<SubscriptionPage />} />
          <Route path="billing" element={<BillingPage />} />
          <Route path="payment-methods" element={<PaymentMethodsPage />} />
          <Route path="profile" element={<ProfilePage />} />
          <Route path="support" element={<SupportPage />} />
        </Route>

        {/* Redirect root to dashboard */}
        <Route path="/" element={<Navigate to="/dashboard" replace />} />
        
        {/* 404 */}
        <Route path="*" element={<div>404 - Page Not Found</div>} />
      </Routes>
    </BrowserRouter>
  );
}

export default App;
